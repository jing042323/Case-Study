sum_impr_output = impr_output.groupby('userId').sum().reset_index(drop=False)

#calculate CTR and assign activity level to each user, 1 for active, 0 for inactive
output_data = sum_impr_output.merge(user_total_impr,how='inner',on='userId')
output_data['ctr'] = output_data['sumClick']/output_data['impression_counts']

mean_ctr = output_data['ctr'].mean()

activity=[]
for x in output_data['ctr']:
    if x >= mean_ctr:
        activity.append(1)
    else:
        activity.append(0)

output_data['activity'] = activity
